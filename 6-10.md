class Solution {
public:
    int f[110][110];

    int uniquePaths(int m, int n) 
    {
        f[0][0]=1;
        
        for(int i=0;i<m;i++)
        {
            for(int j=0;j<n;j++)
            {
                if(i-1>=0&&j-1>=0) f[i][j]=f[i-1][j]+f[i][j-1];
                else if(i-1>=0) f[i][j]=f[i-1][j];
                else if(j-1>=0) f[i][j]=f[i][j-1];
            }
        }

        return f[m-1][n-1];
    }
};
